import src/processes/test/pars.asm

goto main

var dir = "/"

# שימוש במילת המפתח החדשה 'tokens'
tokens cmd = q
tokens args = q

def printDir
    print_raw dir
    print_raw ">"
    ret

def main
    call printDir
    input user_input
    call tokenize_to_stack, user_input

    pop args
    pop cmd

    call dispatcher

    goto main


def dispatcher
    if cmd == "ECHO"
        print args
    else
        print_raw "did not find cmd: "
        print cmd
    endif

    ret
