import src/processes/test/pars.asm
import src/processes/test/std.asm

goto main

var dir = "/"

# שימוש במילת המפתח החדשה 'tokens'
tokens cmd = q
tokens args = q

def printDir
    print_raw dir
    print_raw ">"
    ret

def main
    call printDir
    input user_input
    call tokenize_to_stack, user_input

    pop args
    pop cmd

    call dispatcher

    goto main


def dispatcher
    if cmd == "ECHO"
        print args
    elif cmd == "HELP"
        print "usage: ..."
    elif cmd == "CLEAR"
        call clear_screen
    elif cmd == "DISKTEST"
        call disk_test, dir
    elif cmd == "WRITE"
        call write_to_disk, args
    elif cmd == "LS"
        call ls_disk
    else
        print_raw "did not find cmd: "
        print cmd
    endif

    ret
